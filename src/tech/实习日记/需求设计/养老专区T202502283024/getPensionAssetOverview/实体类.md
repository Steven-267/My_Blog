##  `HoldingItemDto.java`

```java
package com.hundsun.kyp.iae.api.wealthmng.dto.response.scene.pension;

import io.swagger.annotations.ApiModel;
import io.swagger.annotations.ApiModelProperty;
import lombok.Data;

import java.io.Serializable;
import java.math.BigDecimal;

@Data
@ApiModel("资产持仓产品明细")
public class HoldingItemDto implements Serializable {

    @ApiModelProperty("产品名称")
    private String productName;

    @ApiModelProperty("产品代码")
    private String productCode;

    // 通用类字段（理财、基金）
    @ApiModelProperty("产品市值")
    private BigDecimal marketValue;

    @ApiModelProperty("持仓收益")
    private BigDecimal profit;

    @ApiModelProperty("收益率")
    private String profitRate;

    // 存款/国债
    @ApiModelProperty("投本金")
    private BigDecimal principal;

    @ApiModelProperty("到期收益/利息")
    private BigDecimal expectedInterest;

    // 保险
    @ApiModelProperty("投入金额")
    private BigDecimal investedAmount;

    @ApiModelProperty("预计收益")
    private BigDecimal expectedPayout;

    // 公共字段
    @ApiModelProperty("年化收益率")
    private String annualRate;

    @ApiModelProperty("期限")
    private String term;

    @ApiModelProperty("到期日")
    private String maturityDate;
}
```
